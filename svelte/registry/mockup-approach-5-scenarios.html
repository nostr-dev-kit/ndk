<script> import { Relay } from '@nostr-svelte/ui'; import NDK from '@nostr-dev-kit/ndk'; const ndk = new NDK(); const relays = [ 'wss://relay.damus.io', 'wss://nos.lol', 'wss://nostr.wine' ]; </script> <div class="relay-list"> {#each relays as url} <Relay.Root {ndk} {url}> <div class="list-item"> <Relay.Icon size={40} /> <Relay.Header showUrl={true} /> </div> </Relay.Root> {/each} </div> <style> .relay-list { display: flex; flex-direction: column; gap: 0; } .list-item { display: flex; gap: 12px; align-items: center; padding: 12px; border-bottom: 1px solid #eee; } </style>
